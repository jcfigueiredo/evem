// Create ConnectionManager
const connectionManager = new ConnectionManager(evem);

// Setup WebSocket with auto-wiring
const ws = new WebSocket('wss://api.example.com');

ws.onopen = async () => {
    await connectionManager.transitionTo('connected');
};

ws.onclose = async () => {
    await connectionManager.transitionTo('disconnected');
};

// Subscribe to connection state changes
evem.subscribe('ws.connection.state', (event) => {
    console.log(`State: ${event.from} ‚Üí ${event.to}`);
});

// Enable message queue with auto-flush
messageQueue.enable(100, { autoFlush: true });

// Send a message - automatically queued if offline
await evem.publish('ws.send', {
    type: 'message',
    text: 'Hello from EvEm!',
    timestamp: Date.now()
});

// Request-Response pattern
const reqId = 'req-123';
await evem.publish('ws.send', {
    type: 'request',
    id: reqId,
    method: 'getUser',
    params: { id: 123 }
});

// Listen for response
evem.subscribe('ws.response', (message) => {
    if (message.id === reqId) {
        console.log('Response:', message.result);
    }
});
