1. Global Middleware

Intercept all events before they reach handlers

// Global middleware intercepts ALL events
evem.use(async (event, data, next) => {
    console.log('Middleware:', event);
    await next(); // Continue to handlers
});

// Subscribe to some events
evem.subscribe('user.login', (data) => {
    console.log('Login handler:', data);
});

// Publish events - middleware runs first
await evem.publish('user.login', { user: 'Alice' });
await evem.publish('system.update', { version: '2.0' });

2. Pattern-Based Middleware

Apply middleware only to events matching a pattern

// Pattern-based middleware only runs for matching events
evem.use('api.*', async (event, data, next) => {
    data.timestamp = Date.now();
    data.requestId = Math.random().toString(36).substr(2, 9);
    await next();
});

// Subscribe to API events
evem.subscribe('api.*', (data) => {
    console.log('API handler:', data);
});

// These will trigger the middleware
await evem.publish('api.users.get', { userId: 123 });
await evem.publish('api.posts.create', { title: 'Hello' });

3. Event Transformations

Transform event data before it reaches handlers

// Transform modifies data before handler receives it
evem.subscribe('user.input', (data) => {
    console.log('Received:', data.text); // "HELLO WORLD"
}, {
    transform: (data) => ({
        ...data,
        text: data.text.toUpperCase(),
        processed: true
    })
});

// Publish original data
await evem.publish('user.input', {
    text: 'hello world',
    timestamp: Date.now()
});

4. Transform Chains

Multiple transforms applied in sequence

// Each subscriber can have its own transform
// Transforms are applied independently per handler

// Handler 1: trim whitespace
evem.subscribe('data.process', (data) => {
    console.log('Step 1:', data.value); // "hello world"
}, {
    transform: (data) => ({
        ...data,
        value: data.value.trim()
    })
});

// Handler 2: uppercase
evem.subscribe('data.process', (data) => {
    console.log('Step 2:', data.value); // "HELLO WORLD"
}, {
    transform: (data) => ({
        ...data,
        value: data.value.toUpperCase()
    })
});

// Handler 3: add prefix
evem.subscribe('data.process', (data) => {
    console.log('Step 3:', data.value); // "PROCESSED: HELLO WORLD"
}, {
    transform: (data) => ({
        ...data,
        value: `PROCESSED: ${data.value}`
    })
});

await evem.publish('data.process', {
    value: '  hello world  '
});

5. Middleware Cancellation

Middleware can cancel event propagation

// Middleware can cancel events by throwing an error
evem.use('user.delete', async (event, data, next) => {
    if (!data.confirmed) {
        // Cancel event by throwing with 'CANCEL' message
        throw new Error('CANCEL');
    }
    await next(); // Continue to handlers
});

evem.subscribe('user.delete', (data) => {
    console.log('Deleted user:', data.userId);
});

// This will reach the handler
await evem.publish('user.delete', {
    userId: 123,
    confirmed: true
});

// This will be cancelled by middleware
await evem.publish('user.delete', {
    userId: 456,
    confirmed: false
});
