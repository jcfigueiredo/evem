1. Simple Validators

Basic data type and structure validation

// Setup a simple validator for user creation
evem.setSchema('user.create', {
    validate: (data) => {
        if (!data.username) throw new Error('username required');
        if (!data.email) throw new Error('email required');
        if (data.username.length < 3) throw new Error('username must be at least 3 characters');
        return true;
    }
});

// Subscribe to the event
evem.subscribe('user.create', (data) => {
    console.log('User created:', data.username);
});

// Publish valid data
evem.publish('user.create', {
    username: 'alice123',
    email: 'alice@example.com'
});

2. Advanced Validators

Complex validation with detailed error messages

// Advanced multi-field validation
evem.setSchema('payment.process', {
    validate: (data) => {
        const errors = [];

        // Validate amount
        if (!data.amount || typeof data.amount !== 'number' || data.amount <= 0) {
            errors.push('Amount must be a positive number');
        }

        // Validate currency code
        if (!data.currency || !/^[A-Z]{3}$/.test(data.currency)) {
            errors.push('Currency must be a 3-letter code (e.g., USD)');
        }

        // Validate payment method
        if (!data.method || !['card', 'bank', 'crypto'].includes(data.method)) {
            errors.push('Payment method must be: card, bank, or crypto');
        }

        // Throw all errors at once
        if (errors.length > 0) {
            throw new Error(errors.join('; '));
        }

        return true;
    }
});

3. Error Policies

Configure how validation errors are handled

Current Policy: throw 
// Set error policy to control validation failure behavior
evem.setErrorPolicy('throw');  // Stop execution and throw error (default)
evem.setErrorPolicy('log');    // Log error to console and skip event
evem.setErrorPolicy('ignore'); // Silently skip invalid events

// Example: Using 'log' policy
evem.setErrorPolicy('log');

evem.setSchema('test.event', {
    validate: (data) => {
        if (!data.valid) throw new Error('Data is invalid');
        return true;
    }
});

// This will log an error but won't throw
evem.publish('test.event', { valid: false });

4. Type Safety

Validate data types and structure

// Type checking validation
evem.setSchema('data.submit', {
    validate: (data) => {
        const errors = [];

        if (typeof data.id !== 'number') {
            errors.push('id must be a number');
        }

        if (typeof data.name !== 'string') {
            errors.push('name must be a string');
        }

        if (!Array.isArray(data.tags)) {
            errors.push('tags must be an array');
        }

        if (typeof data.active !== 'boolean') {
            errors.push('active must be a boolean');
        }

        if (errors.length > 0) {
            throw new Error(errors.join('; '));
        }

        return true;
    }
});

// Valid data
evem.publish('data.submit', {
    id: 123,
    name: 'Test Item',
    tags: ['tag1', 'tag2'],
    active: true
});

5. Custom Validators

Create reusable validation functions

// Create reusable validator functions
const validators = {
    email: (value) => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value),
    url: (value) => /^https?:\/\/.+/.test(value),
    phone: (value) => /^\+?[\d\s-()]+$/.test(value),
    notEmpty: (value) => value && value.length > 0
};

// Use them in your schema
evem.setSchema('contact.add', {
    validate: (data) => {
        const errors = [];

        if (!validators.notEmpty(data.name)) {
            errors.push('Name cannot be empty');
        }

        if (!validators.email(data.email)) {
            errors.push('Invalid email format');
        }

        if (data.website && !validators.url(data.website)) {
            errors.push('Invalid website URL');
        }

        if (data.phone && !validators.phone(data.phone)) {
            errors.push('Invalid phone format');
        }

        if (errors.length > 0) {
            throw new Error(errors.join('; '));
        }

        return true;
    }
});
