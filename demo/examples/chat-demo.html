// 1. Setup: Initialize EvEm and core components
const evem = new EvEm();
const connectionManager = new ConnectionManager(evem);
const messageQueue = new MessageQueue(evem, connectionManager);

// 2. Connection Manager: Handles WebSocket state transitions
class ConnectionManager {
    constructor(evem) {
        this.evem = evem;
        this.state = 'disconnected';
        this.ws = null;
    }

    async transitionTo(newState) {
        const oldState = this.state;
        this.state = newState;

        // Publish state change event
        await this.evem.publish('ws.connection.state', {
            from: oldState,
            to: newState,
            timestamp: Date.now()
        });
    }

    isConnected() {
        return this.state === 'connected';
    }

    connect(url) {
        this.ws = new WebSocket(url);
        this.transitionTo('connecting');

        this.ws.onopen = () => {
            this.transitionTo('connected');
        };

        this.ws.onclose = () => {
            this.transitionTo('disconnected');
        };

        this.ws.onmessage = (event) => {
            const data = JSON.parse(event.data);
            this.evem.publish('server.message', data);
        };
    }

    disconnect() {
        if (this.ws) {
            this.ws.close();
        }
    }
}

// 3. Message Queue: Queues messages when offline
class MessageQueue {
    constructor(evem, connectionManager) {
        this.evem = evem;
        this.connectionManager = connectionManager;
        this.queue = [];
    }

    enable() {
        // Queue messages when offline
        this.evem.subscribe('ws.send', (data) => {
            if (!this.connectionManager.isConnected()) {
                this.queue.push(data);
                this.evem.publish('message.queued', {
                    queueSize: this.queue.length
                });
            } else {
                this.evem.publish('ws.send.queued', data);
            }
        });

        // Flush queue when reconnected
        this.evem.subscribe('ws.connection.state', async (event) => {
            if (event.to === 'connected' && this.queue.length > 0) {
                await this.flush();
            }
        });
    }

    async flush() {
        const messages = [...this.queue];
        this.queue = [];

        for (const msg of messages) {
            await this.evem.publish('ws.send.queued', msg);
            // Small delay between messages
            await new Promise(r => setTimeout(r, 100));
        }

        this.evem.publish('queue.flushed', {
            messageCount: messages.length
        });
    }

    getQueueSize() {
        return this.queue.length;
    }
}

    }
});

    displayMessage(data.user, data.text, data.timestamp);
});

// Handle typing indicators

    showTypingIndicator(data.user);
});

    await evem.publish('ws.send', {
        type: 'message',
        user: currentUser,
        text: text,
        timestamp: Date.now()
    });
